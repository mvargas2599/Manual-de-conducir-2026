const contenedor = document.getElementById('preguntas-contenedor');

bancoPreguntas.forEach((p, i) => {
    let opcionesTexto = p.options.join(". ");
    let textoCompleto = `${p.q} Opciones: ${opcionesTexto}`;
    
    let html = `
    <div class="pregunta-bloque">
        <button type="button" class="btn-audio" onclick="leer('${textoCompleto.replace(/'/g, "")}')">🔊 Escuchar</button>
        <span class="pregunta-texto">${p.q}</span>
        <div class="opciones">`;
    
    p.options.forEach((opt, index) => {
        html += `<label><input type="radio" name="p${i}" value="${index}"> ${opt}</label>`;
    });
    
    html += `</div><div id="res-p${i}" class="feedback" style="display:none;"></div></div>`;
    contenedor.innerHTML += html;
});

function leer(texto) {
    window.speechSynthesis.cancel(); // Detiene cualquier audio previo
    const lectura = new SpeechSynthesisUtterance(texto);
    lectura.lang = 'es-US';
    lectura.rate = 0.9; // Velocidad un poco más lenta para claridad
    window.speechSynthesis.speak(lectura);
}

function calificar() {
    let score = 0;
    bancoPreguntas.forEach((p, i) => {
        const seleccion = document.querySelector(`input[name="p${i}"]:checked`);
        const feedback = document.getElementById(`res-p${i}`);
        feedback.style.display = "block";
        
        if (seleccion && parseInt(seleccion.value) === p.a) {
            score++;
            feedback.innerHTML = "✓ Correcta";
            feedback.className = "feedback correcta";
        } else {
            feedback.innerHTML = "✗ Incorrecta. La respuesta correcta era: " + p.options[p.a];
            feedback.className = "feedback incorrecta";
        }
    });
    
    const resultDiv = document.getElementById('resultado');
    resultDiv.style.display = "block";
    resultDiv.innerHTML = "Puntaje final: " + score + " / " + bancoPreguntas.length;
    window.scrollTo(0, 0);
}
